   cd ~/projects
   git clone <my-grid-url> my-grid
   git clone <atari-style-url> atari-style

2. Setup atari-style:
   cd ~/projects/atari-style
   python -m venv venv
   source venv/Scripts/activate
   pip install -r requirements.txt

3. Test shader output:
   python -m atari_style.zone_mode plasma --width 40 --height 20 --fps 10
   # You should see colorful animated plasma! Press Ctrl+C to stop

4. Setup my-grid:
   cd ~/projects/my-grid
   pip install -r requirements.txt

5. Test basic my-grid:
   python mygrid.py
   # Press 'q' to quit

6. Test with server API:
   python mygrid.py --server --port 8765
   # In another terminal:
   echo ":text Hello Windows!" | nc localhost 8765
